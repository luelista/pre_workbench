pcap_file variant {
	struct (endianness="<", section="pcap file, little endian"){
		header pcap_header
		packets repeat pcap_packet
	}
	struct (endianness=">", section="pcap file, big endian"){
		header pcap_header
		packets repeat pcap_packet
	}
	repeat(endianness="<", section="pcapNG file, little endian") pcapng_block
	repeat(endianness=">", section="pcapNG file, big endian") pcapng_block
}

pcap_header struct (section="pcap file header"){
	magic_number uint32(description="'A1B2C3D4' means the endianness is correct", magic=2712847316)
	version_major uint16(description="major number of the file format")
	version_minor uint16(description="minor number of the file format")
	thiszone int32(description="correction time in seconds from UTC to local time (0)")
	sigfigs uint32(description="accuracy of time stamps in the capture (0)")
	snaplen uint32(description="max length of captured packed (65535)")
	encap_proto uint32(description="type of data link (1 = ethernet)")
}

pcap_packet struct {
	pheader struct (section="pcap packet header"){
		ts_sec uint32(description="timestamp seconds")
		ts_usec uint32(description="timestamp microseconds")
		incl_len uint32(description="number of octets of packet saved in file")
		orig_len uint32(description="actual length of packet")
	}
	payload bytes(size_expr=(pheader.incl_len), parse_with=ll_protos)
}

ll_protos switch header.encap_proto (endianness=">"){
	case 1: ether
	case 128: llzp
}

nl_protos switch eheader.ethertype {
	case 0x0800: ip4_packet
	case 0x0806: arp
}

mac_addr bytes(size_expr="6")

ip4_addr bytes(size_expr="4")

ip4_packet struct {
	ipheader struct (section="IPv4 header"){
		version_hlen uint8
		tos uint8
		total_length uint16
		identification uint16
		flags_fragoffset uint16
		ttl uint8
		protocol uint8
		cksum uint16
		src_ip ip4_addr
		dst_ip ip4_addr
	}
	payload ip_payload
}

ip_payload switch ipheader.protocol {
	case 6: tcp_segment
	case 17: udp_segment
}

tcp_segment struct {
	tcpheader struct (section="TCP Header"){
		src_port uint16
		dst_port uint16
		seq_num uint32
		ack_num uint32
		flags uint16
		window_size uint16
		cksum uint16
		urg_ptr uint16
		options bytes(size_expr=((((flags & 0xf000) >> 12) * 4) - 20))
	}
	tcp_payload bytes(section="TCP Payload", color="#00aa00")
}

udp_segment struct {
}

arp struct {
}

ether struct (endianness=">"){
	eheader struct (section="ethernet header"){
	dst_addr mac_addr
	src_addr mac_addr
	ethertype uint16
}
	payload nl_protos
}

llzp struct {
	header struct {
	version uint8
	type uint8
	encap_proto uint16
	fields repeat(until=(_[-1].tag == 1)) variant {
			struct {
				tag uint8(magic=0)
			}
			struct {
				tag uint8(magic=1)
			}
			struct {
				tag uint8
				length uint8
				body bytes(size_expr="length")
			}
		}
}
	payload ll_protos
}

pcapng_block struct (section="pcapNG block"){
	block_type fixed(format="I", color="#999900")
	block_length fixed(format="I", color="#666600")
	block_payload bytes(size_expr="block_length-12", parse_with=pcapng_block_payload)
	block_length2 fixed(format="I", color="#666600")
}

pcapng_block_payload switch block_type {
	case 0x0A0D0D0A: pcapng_SHB
	case 1: pcapng_IDB
	case 6: pcapng_EPB
}

pcapng_SHB struct {
	byte_order_magic fixed(format="I", magic=439041101, color="green")
	version_major fixed(format="H")
	version_minor fixed(format="H")
	section_length fixed(format="q")
	options pcapng_options
}

pcapng_IDB struct {
	linktype fixed(format="H")
	reserved fixed(format="H")
	snaplen fixed(format="I")
	options pcapng_options
}

pcapng_EPB struct {
	interface_id fixed(format="I")
	timestamp fixed(format="Q")
	cap_length fixed(format="I")
	orig_length fixed(format="I")
	payload bytes(size_expr="cap_length", parse_with=ether)
	payload_padding bytes(size_expr="3-((cap_length-1)&3)", textcolor="#888888")
}

pcapng_options repeat struct {
		code fixed(format="H")
		length fixed(format="H")
		value bytes(size_expr="length")
	}

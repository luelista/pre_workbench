variant {
	[endianness="<"] struct {
		file_header struct {
			magic_number [format="I", description="'A1B2C3D4' means the endianness is correct", magic=2712847316] uint32
			version_major [format="H", description="major number of the file format"] uint16
			version_minor [format="H", description="minor number of the file format"] uint16
			thiszone [format="i", description="correction time in seconds from UTC to local time (0)"] int32
			sigfigs [format="I", description="accuracy of time stamps in the capture (0)"] uint32
			snaplen [format="I", description="max length of captured packed (65535)"] uint32
			network [format="I", description="type of data link (1 = ethernet)"] uint32
		}
		packets repeat("*"){
			struct {
				header struct {
					ts_sec [format="I", description="timestamp seconds"] uint32
					ts_usec [format="I", description="timestamp microseconds"] uint32
					incl_len [format="I", description="number of octets of packet saved in file"] uint32
					orig_len [format="I", description="actual length of packet"] uint32
				}
				payload [size_expr="header.incl_len"] bytes
			}
}
	}

		}
		packets repeat("*"){
			struct {
				header struct {
					ts_sec [format="I", description="timestamp seconds"] uint32
					ts_usec [format="I", description="timestamp microseconds"] uint32
					incl_len [format="I", description="number of octets of packet saved in file"] uint32
					orig_len [format="I", description="actual length of packet"] uint32

				payload [size_expr="header.incl_len"] bytes
			}
}
	}
